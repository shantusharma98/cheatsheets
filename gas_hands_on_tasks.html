Scenario: Your manager dumps raw sales data into a Google Sheet every Monday. Your job: write a script that automatically highlights overdue deals, marks high-value deals in gold, and flags deals with no follow-up date.

function formatSalesReport() {

  // Get the active spreadsheet and the "Sales" sheet tab
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName("Sales");

  // Get all data rows (starting from row 2, skip header)
  const lastRow = sheet.getLastRow();
  const today = new Date();

  // First clear all existing background colours so we start fresh
  sheet.getRange(2, 1, lastRow - 1, 5).setBackground(null);

  // Loop through each data row (row 2 onwards)
  for (let i = 2; i <= lastRow; i++) {

    // getRange(row, column) — row i, columns 1-5
    const row = sheet.getRange(i, 1, 1, 5);
    const values = row.getValues()[0]; // [0] because getValues returns 2D array

    const dealValue  = values[2]; // Column C (index 2)
    const status     = values[3]; // Column D (index 3)
    const deadline   = values[4]; // Column E (index 4)

    // RULE 1: Won deals → light green background
    if (status === "Won") {
      row.setBackground("#d9ead3"); // light green
      continue; // skip other rules for this row
    }

    // RULE 2: No deadline set → orange background (needs attention)
    if (!deadline || deadline === "") {
      row.setBackground("#fce5cd"); // light orange
      continue;
    }

    // RULE 3: Deadline is in the past → red (overdue)
    const deadlineDate = new Date(deadline);
    if (deadlineDate < today && status !== "Won") {
      row.setBackground("#f4cccc"); // light red
      continue;
    }

    // RULE 4: High value deal (>= 200000) → gold/yellow
    if (dealValue >= 200000) {
      row.setBackground("#fff2cc"); // light yellow/gold
    }
  }

  // Log a message to the execution log so you can confirm it ran
  Logger.log("Formatting done! Rows processed: " + (lastRow - 1));
}

function sendMailMerge() {

  // Get the "Leads" sheet tab
  const sheet = SpreadsheetApp
    .getActiveSpreadsheet()
    .getSheetByName("Leads");

  // Get all data as a 2D array. Row 0 = headers, Row 1+ = data
  const data = sheet.getDataRange().getValues();

  let emailsSent = 0;

  // Start from index 1 to skip the header row
  for (let i = 1; i < data.length; i++) {

    // Destructure: pull column values into named variables
    const [name, email, company, status] = data[i];

    // Only send if status is "New" — skip already-emailed rows
    if (status !== "New") continue;

    // Build the personalised email subject and body
    const subject = `Welcome to our platform, ${name}!`;

    const htmlBody = `
      <div style="font-family: Arial, sans-serif; max-width: 500px;">
        <h2 style="color: #1a7a4a;">Hi ${name} 👋</h2>
        <p>Thank you for your interest from <strong>${company}</strong>.</p>
        <p>We're excited to work with you! Our team will reach out 
           within 24 hours to schedule a demo.</p>
        <p style="color: #666;">— The Automation Team</p>
      </div>
    `;

    // Send the email using GmailApp (no setup needed, uses your account)
    GmailApp.sendEmail(
      email,    // to
      subject,  // subject
      "",       // plain text body (empty, using HTML instead)
      { htmlBody: htmlBody }  // options object with HTML version
    );

    // Update status in column D (column index 4 = 1-based) to "Emailed"
    // i+1 because sheet rows are 1-based, but our array is 0-based
    sheet.getRange(i + 1, 4).setValue("Emailed");

    emailsSent++;
    Logger.log(`Sent email to: ${name} <${email}>`);
  }

  Logger.log(`Done! Total emails sent: ${emailsSent}`);

  // Show a popup in the Sheet to confirm
  SpreadsheetApp.getUi().alert(`✅ Sent ${emailsSent} emails!`);
}
✓ Expected result
  • Emails go to rows 1, 2, 4 (status = "New"). Row 3 (Sneha, already "Emailed") is skipped.
  • Column D for rows 1, 2, 4 updates to "Emailed" in the sheet
  • A popup says "✅ Sent 3 emails!"
  • Check your inbox — you'll receive 3 nicely formatted HTML emails

// This function is called automatically when a Form is submitted.
// The parameter "e" is the event object — it contains all submitted data.

function onFormSubmit(e) {

  // e.namedValues gives us an object like:
  // { "Your Name": ["Priya"], "Your Email": ["priya@co.com"], ... }
  // The [0] at the end gets the first (only) value from each array
  const name    = e.namedValues["Your Name"][0];
  const email   = e.namedValues["Your Email"][0];
  const company = e.namedValues["Company Name"][0];

  // Build confirmation email
  const subject = `We received your inquiry, ${name}!`;
  const body = `
Hi ${name},

Thank you for reaching out from ${company}!

We have received your inquiry and one of our team members 
will contact you within 1 business day.

Reference ID: ${new Date().getTime()}

Best regards,
The Team
  `;

  // Send the thank-you email to the person who submitted the form
  GmailApp.sendEmail(email, subject, body);

  // Also notify your own team (notify the script owner)
  const teamEmail = Session.getActiveUser().getEmail();
  GmailApp.sendEmail(
    teamEmail,
    `New form submission from ${name} @ ${company}`,
    `Name: ${name}\nEmail: ${email}\nCompany: ${company}`
  );

  Logger.log(`Acknowledgement sent to: ${email}`);
}

Task 4 — Daily Report Email at 9am

Scenario: Every morning at 9am, a script reads your sales sheet, calculates total pipeline value, counts deals by status, and emails a formatted summary to your manager — fully automatic, no button press needed.

Step 1 — Use the Sales sheet from Task 1
📋
This script reads the Sales sheet tab. If you did Task 1, it's already set up. If not, create it with the same columns: Deal Name, Sales Rep, Value, Status, Deadline.
Step 2 — Paste this code
Apps Script — Code.gs 
function sendDailyReport() {

  const sheet = SpreadsheetApp
    .getActiveSpreadsheet()
    .getSheetByName("Sales");

  const data = sheet.getDataRange().getValues();
  const today = new Date();

  // Initialise counters
  let totalPipeline = 0;
  let wonDeals      = 0;
  let inProgress    = 0;
  let overdueDeals  = 0;
  let wonValue      = 0;

  // Loop from row 1 (skip row 0 = headers)
  for (let i = 1; i < data.length; i++) {
    const [deal, rep, value, status, deadline] = data[i];
    if (!deal) continue; // skip empty rows

    totalPipeline += Number(value);

    if (status === "Won") {
      wonDeals++;
      wonValue += Number(value);
    }
    if (status === "In Progress") inProgress++;

    if (deadline && new Date(deadline) < today && status !== "Won") {
      overdueDeals++;
    }
  }

  // Format numbers as Indian Rupees
  const formatINR = (n) => `₹${n.toLocaleString('en-IN')}`;

  // Build the date string for the email subject
  const dateStr = Utilities.formatDate(
    today, 
    Session.getScriptTimeZone(), 
    "dd MMM yyyy"
  );

  const subject = `📊 Daily Sales Report — ${dateStr}`;

  const htmlBody = `
    <div style="font-family:Arial,sans-serif;max-width:520px;border:1px solid #ddd;border-radius:8px;overflow:hidden">
      <div style="background:#1a7a4a;padding:20px 24px">
        <h2 style="color:white;margin:0">Daily Sales Report</h2>
        <p style="color:rgba(255,255,255,0.7);margin:4px 0 0">${dateStr}</p>
      </div>
      <div style="padding:20px 24px">
        <table style="width:100%;border-collapse:collapse">
          <tr><td style="padding:10px;background:#f5f5f5;font-weight:bold">Total Pipeline</td>
              <td style="padding:10px;background:#f5f5f5;text-align:right;font-size:18px;color:#1a7a4a">
                <strong>${formatINR(totalPipeline)}</strong></td></tr>
          <tr><td style="padding:10px">Won Value</td>
              <td style="padding:10px;text-align:right">${formatINR(wonValue)}</td></tr>
          <tr><td style="padding:10px;background:#f5f5f5">Won Deals</td>
              <td style="padding:10px;background:#f5f5f5;text-align:right">${wonDeals}</td></tr>
          <tr><td style="padding:10px">In Progress</td>
              <td style="padding:10px;text-align:right">${inProgress}</td></tr>
          <tr><td style="padding:10px;background:#fce5cd;color:#e65100;font-weight:bold">⚠ Overdue Deals</td>
              <td style="padding:10px;background:#fce5cd;text-align:right;color:#e65100;font-weight:bold">${overdueDeals}</td></tr>
        </table>
      </div>
      <div style="background:#f5f5f5;padding:12px 24px;font-size:12px;color:#999">
        Auto-generated by Google Apps Script. Do not reply.
      </div>
    </div>
  `;

  // Send to yourself (change this to manager's email in real use)
  const recipient = Session.getActiveUser().getEmail();
  GmailApp.sendEmail(recipient, subject, "", { htmlBody: htmlBody });
  Logger.log("Report sent to: " + recipient);
}

// Run this ONCE to set up the daily 9am trigger
function createDailyTrigger() {
  ScriptApp.newTrigger("sendDailyReport")
    .timeBased()
    .everyDays(1)
    .atHour(9)  // 9am in your timezone
    .create();
  Logger.log("Daily trigger created! sendDailyReport will run at 9am every day.");
}

Task 5 — Fetch Live Data from an External API

Scenario: Your team tracks USD→INR exchange rate in a Google Sheet. Instead of manually updating it, a script calls a public API and writes the latest rate to a cell automatically every hour.

Step 1 — Set up a sheet called "Live Data"
📋
Create a new tab called Live Data. Put these labels in column A:
A1: USD to INR Rate  |  A2: Last Updated  |  A3: EUR to INR Rate
Step 2 — Paste this code
Apps Script — Code.gs 
function fetchExchangeRates() {

  const sheet = SpreadsheetApp
    .getActiveSpreadsheet()
    .getSheetByName("Live Data");

  // UrlFetchApp.fetch() = Python's requests.get()
  // This API is free and requires no API key
  const apiUrl = "https://open.er-api.com/v6/latest/USD";

  try {
    // Make the GET request
    const response = UrlFetchApp.fetch(apiUrl);

    // getContentText() = .text in Python's requests
    // JSON.parse() = json.loads() in Python
    const json = JSON.parse(response.getContentText());

    // Navigate the JSON object to get specific rates
    const usdToInr = json.rates.INR;  // e.g. 83.45
    const eurToInr = json.rates.INR / json.rates.EUR;

    // Write rates to column B (column index 2)
    sheet.getRange("B1").setValue(usdToInr.toFixed(2));
    sheet.getRange("B3").setValue(eurToInr.toFixed(2));

    // Write last updated timestamp
    const now = new Date();
    const timestamp = Utilities.formatDate(
      now,
      Session.getScriptTimeZone(),
      "dd MMM yyyy, hh:mm a"
    );
    sheet.getRange("B2").setValue(timestamp);

    Logger.log(`Rates updated. USD→INR: ${usdToInr}`);

  } catch (error) {
    // Always wrap API calls in try/catch — APIs can go down
    Logger.log("Error fetching rates: " + error.message);
  }
}

// Run this once to make it refresh every hour automatically
function createHourlyTrigger() {
  ScriptApp.newTrigger("fetchExchangeRates")
    .timeBased()
    .everyHours(1)
    .create();
  Logger.log("Hourly trigger created!");
}
✓ Expected result
  • B1 shows the live USD→INR rate (e.g. 83.45)
  • B2 shows the exact time the data was last fetched
  • B3 shows the EUR→INR rate
  • After running createHourlyTrigger(), B1-B3 update automatically every hour

// onEdit(e) is a SIMPLE TRIGGER — GAS calls it automatically on every edit.
// No setup in the Triggers menu required. Just name it "onEdit".

function onEdit(e) {

  // e.range = the cell that was just edited
  // e.value = the new value that was typed
  const range = e.range;
  const sheet = range.getSheet();

  // Only run on the "Tasks" sheet tab (ignore other tabs)
  if (sheet.getName() !== "Tasks") return;

  // Only run if the edited cell is in column C (column index 3)
  if (range.getColumn() !== 3) return;

  // Skip the header row
  if (range.getRow() === 1) return;

  const status = e.value; // what was just typed
  const row    = range.getRow();

  // Get the full row range (columns A to D)
  const fullRow = sheet.getRange(row, 1, 1, 4);

  // Apply colour based on status value
  switch (status) {
    case "Done":
      fullRow.setBackground("#d9ead3"); // green
      fullRow.setFontColor("#1a7a4a");
      break;
    case "In Progress":
      fullRow.setBackground("#fff2cc"); // yellow
      fullRow.setFontColor("#92540a");
      break;
    case "Blocked":
      fullRow.setBackground("#f4cccc"); // red
      fullRow.setFontColor("#8a1a1a");
      break;
    default:
      // Any other status (e.g. "Not Started") → clear formatting
      fullRow.setBackground(null);
      fullRow.setFontColor("#000000");
  }
}

function createClientFolders() {

  const sheet = SpreadsheetApp
    .getActiveSpreadsheet()
    .getSheetByName("Clients");

  const data = sheet.getDataRange().getValues();

  // Create or find the parent "Clients" folder in Drive root
  // We check if it already exists to avoid duplicates
  let parentFolder;
  const existingFolders = DriveApp.getFoldersByName("All Clients");

  if (existingFolders.hasNext()) {
    parentFolder = existingFolders.next(); // Use existing folder
  } else {
    parentFolder = DriveApp.createFolder("All Clients"); // Create new
  }

  let created = 0;

  // Loop from row 1 (skip header)
  for (let i = 1; i < data.length; i++) {
    const [clientName, email, folderCreated] = data[i];

    if (!clientName) continue;              // skip blank rows
    if (folderCreated === "Yes") continue; // already done, skip

    // Create subfolder inside "All Clients" with the client's name
    const clientFolder = parentFolder.createFolder(clientName);

    // Create standard subfolders inside each client folder
    clientFolder.createFolder("Documents");
    clientFolder.createFolder("Invoices");
    clientFolder.createFolder("Proposals");

    // Get the shareable URL of the new folder
    const folderUrl = clientFolder.getUrl();

    // Update the sheet: mark as created and write the folder link
    // i+1 = sheet row number (1-based), column 3 = "Folder Created?"
    sheet.getRange(i + 1, 3).setValue("Yes");
    sheet.getRange(i + 1, 4).setValue(folderUrl);

    // Make folder link clickable using a hyperlink formula
    sheet.getRange(i + 1, 4).setFormula(
      `=HYPERLINK("${folderUrl}","Open Folder")`
    );

    created++;
    Logger.log(`Created folder for: ${clientName}`);
  }

  SpreadsheetApp.getUi().alert(`✅ Created ${created} client folders!`);
}
✓ Expected result
  • A folder called "All Clients" appears in your Google Drive root
  • Inside it: Acme Corp, TechStart India, Mega Bank — each with Documents / Invoices / Proposals subfolders
  • Column C updates to "Yes", Column D becomes a clickable "Open Folder" link
  • Run the script again — it skips already-created folders (won't duplicate)

// onOpen runs automatically when the spreadsheet opens.
// It adds a custom "Invoice Tools" menu to the top menu bar.
function onOpen() {
  SpreadsheetApp.getUi()
    .createMenu("📄 Invoice Tools")
    .addItem("Generate & Send Invoice", "generateAndSendInvoice")
    .addToUi();
}

function generateAndSendInvoice() {

  const ss    = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName("Invoices");

  // Read which row number to generate invoice for (from cell I1)
  const targetRow = sheet.getRange("I1").getValue();
  const rowData   = sheet.getRange(targetRow, 1, 1, 6).getValues()[0];

  const [client, email, amount, invoiceNo, description, dueDate] = rowData;

  // Format amount as Indian Rupees
  const formattedAmount = `₹${Number(amount).toLocaleString('en-IN')}`;

  // ---- CREATE GOOGLE DOC INVOICE ----
  const doc  = DocumentApp.create(`Invoice ${invoiceNo} — ${client}`);
  const body = doc.getBody();

  // Set page margins
  body.setMarginTop(50).setMarginBottom(50)
      .setMarginLeft(60).setMarginRight(60);

  // --- Header ---
  const header = body.appendParagraph("TAX INVOICE");
  header.setHeading(DocumentApp.ParagraphHeading.HEADING1);
  header.setAlignment(DocumentApp.HorizontalAlignment.CENTER);

  body.appendParagraph("\nYour Company Name\ninfo@yourcompany.com | +91 98765 43210")
      .setAlignment(DocumentApp.HorizontalAlignment.CENTER);

  body.appendHorizontalRule();

  // --- Invoice Details ---
  body.appendParagraph(`Invoice No: ${invoiceNo}`).setBold(true);
  body.appendParagraph(`Date: ${new Date().toDateString()}`);
  body.appendParagraph(`Due Date: ${new Date(dueDate).toDateString()}`);
  body.appendParagraph(`\nBilled To:\n${client}\n${email}`);
  body.appendHorizontalRule();

  // --- Item Table ---
  const table = body.appendTable([
    ["Description",   "Amount"       ],
    [description,     formattedAmount],
    ["TOTAL",          formattedAmount]
  ]);

  // Style header row of table
  table.getRow(0).editAsText().setBold(true);
  table.getRow(2).editAsText().setBold(true);

  body.appendParagraph("\nThank you for your business!")
      .setAlignment(DocumentApp.HorizontalAlignment.CENTER);

  doc.saveAndClose();

  // ---- CONVERT DOC TO PDF ----
  // getAs('application/pdf') converts the Google Doc to PDF bytes
  const docFile = DriveApp.getFileById(doc.getId());
  const pdfBlob = docFile.getAs("application/pdf");
  pdfBlob.setName(`${invoiceNo}_${client}.pdf`);

  // ---- EMAIL THE PDF ----
  GmailApp.sendEmail(
    email,
    `Invoice ${invoiceNo} from Your Company`,
    `Dear ${client},\n\nPlease find your invoice attached.\n\nAmount Due: ${formattedAmount}\nDue Date: ${dueDate}\n\nThank you,\nYour Company`,
    {
      attachments: [pdfBlob],
      htmlBody: `<p>Dear <strong>${client}</strong>,</p><p>Please find your invoice attached.</p><p>Amount Due: <strong>${formattedAmount}</strong><br>Due: ${dueDate}</p><p>Thank you!</p>`
    }
  );

  // Clean up: delete the intermediate Google Doc from Drive
  docFile.setTrashed(true);

  SpreadsheetApp.getUi().alert(`✅ Invoice ${invoiceNo} sent to ${email}!`);
}
